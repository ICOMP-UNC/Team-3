27void sys_tick_handler(void)
28{
29 systick_get_countflag();
31 if (button_get_stop_flag()) // Stopped
32 {
33 systick_counter_disable();
34 lcd_clear();
35 lcd_print_string(" STOPPED! ");
36 lcd_set_cursor(2, 0);
37 lcd_print_string("Please Restart ");
38 }
39 else // Not Stopped
40 {
41 if (button_get_object_flag()) // Object
42 {
43 motor_disable();
44 if (response__measurement_count >= MEASUREMENT_RATE)
45 {
46 response__measurement_count = 0;
47 measure();
48 }
49 if (response__display_count >= DISPLAY_RATE / 2)
50 {
51 response__display_count = 0;
52 display_measure_info();
53 }
54 response__measurement_count++;
55 response__display_count++;
56 }
57 else // No object
58 {
59 motor_enable();
60 response__measurement_count = 0;
61 if (response__pid_count >= PID_RATE)
62 {
63 response__pid_count = 0;
64 upt_pid();
65 }
66 if (response__display_count >= DISPLAY_RATE)
67 {
68 response__display_count = 0;
69 if (showing_measure_flag) // Display Measurement
70 {
71 if (remaining_measure_dtime > 0)
72 {
73 remaining_measure_dtime--;
74 }
75 else
76 {
77 showing_measure_flag = 0;
78 remaining_measure_dtime = MEASUREMENT_DISPLAY_TIME;
79 }
80 }
81 else
82 { // Display Speed
83 display_speed();
84 }
85 }
86 response__pid_count++;
87 response__display_count++;
88 }
89 }
